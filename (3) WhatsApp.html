public int findStringCode(String input1) {     String[] words = input1.toUpperCase().split(" ");     StringBuilder result = new StringBuilder();     for (String word : words) {         int sum = 0;         for (int i = 0; i < word.length() / 2; i++) {             char startChar = word.charAt(i);             char endChar = word.charAt(word.length() - 1 - i);             sum += Math.abs(startChar - 'A' + 1 - (endChar - 'A' + 1));         }         result.append(sum);     }     return Integer.parseInt(result.toString()); }
